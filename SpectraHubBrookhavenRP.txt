local function FireRemote(remoteName, ...)
    local remote = RE:FindFirstChild(remoteName)
    if remote then
        pcall(function() remote:FireServer(...) end)
    end
end

local function InvokeRemote(remoteName, ...)
    local remote = RE:FindFirstChild(remoteName)
    if remote then
        return pcall(function() return remote:InvokeServer(...) end)
    end
    return false
end

local function GetESPColor()
    if _G.ESPData.selectedColor == "RGB" then
        return Color3.fromHSV(tick() % 5 / 5, 1, 1)
    end
    local colors = {
        Branco = Color3.fromRGB(255, 255, 255),
        Preto = Color3.fromRGB(0, 0, 0),
        Vermelho = Color3.fromRGB(255, 0, 0),
        Verde = Color3.fromRGB(0, 255, 0),
        Azul = Color3.fromRGB(0, 170, 255),
        Amarelo = Color3.fromRGB(255, 255, 0),
        Rosa = Color3.fromRGB(255, 105, 180),
        Roxo = Color3.fromRGB(128, 0, 128)
    }
    return colors[_G.ESPData.selectedColor] or Color3.new(1, 1, 1)
end

_G.clearAllESP = function()
    for _, gui in pairs(_G.ESPData.billboardGuis) do
        pcall(function() gui:Destroy() end)
    end
    for _, highlight in pairs(_G.ESPData.highlights) do
        pcall(function() highlight:Destroy() end)
    end
    for _, line in pairs(_G.ESPData.lines) do
        pcall(function() line:Remove() end)
    end
    for _, conn in pairs(_G.ESPData.connections) do
        pcall(function() conn:Disconnect() end)
    end
    _G.ESPData.billboardGuis = {}
    _G.ESPData.highlights = {}
    _G.ESPData.lines = {}
    _G.ESPData.connections = {}
end

local function CreateBillboardESP(player)
    if player == LocalPlayer then return end
    if not player.Character then return end
    if not player.Character:FindFirstChild("Head") then return end
    local gui = Instance.new("BillboardGui")
    gui.Size = UDim2.new(0, 200, 0, 50)
    gui.StudsOffset = Vector3.new(0, 3, 0)
    gui.AlwaysOnTop = true
    gui.Parent = player.Character.Head
    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(1, 0, 1, 0)
    label.BackgroundTransparency = 1
    label.TextStrokeTransparency = 0.5
    label.Font = Enum.Font.SourceSansBold
    label.TextSize = 14
    label.Text = player.Name .. " | " .. player.AccountAge .. " dias"
    label.TextColor3 = GetESPColor()
    label.Parent = gui
    _G.ESPData.billboardGuis[player] = gui
end

local function CreateHighlightESP(player)
    if player == LocalPlayer then return end
    if not player.Character then return end
    local highlight = Instance.new("Highlight")
    highlight.FillColor = GetESPColor()
    highlight.FillTransparency = 0.5
    highlight.OutlineTransparency = 0
    highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    highlight.Parent = player.Character
    _G.ESPData.highlights[player] = highlight
end

local function CreateLineESP(player)
    if player == LocalPlayer then return end
    local line = Drawing.new("Line")
    line.Thickness = 2
    line.Transparency = 1
    line.Visible = false
    _G.ESPData.lines[player] = line
end

_G.applyCurrentESP = function()
    _G.clearAllESP()
    if not _G.ESPData.espEnabled then return end
    for _, player in pairs(Players:GetPlayers()) do
        if _G.ESPData.espType == "Nome + Idade" then
            CreateBillboardESP(player)
        elseif _G.ESPData.espType == "Corpo (Highlight)" then
            CreateHighlightESP(player)
        elseif _G.ESPData.espType == "Linhas" then
            CreateLineESP(player)
        end
    end
    table.insert(_G.ESPData.connections, RunService.RenderStepped:Connect(function()
        if not _G.ESPData.espEnabled then return end
        GetESPColor()
        if _G.ESPData.espType == "Linhas" then
            for player, line in pairs(_G.ESPData.lines) do
                local char = player.Character
                if char and char:FindFirstChild("HumanoidRootPart") then
                    local pos, onScreen = CurrentCamera:WorldToViewportPoint(char.HumanoidRootPart.Position)
                    if onScreen then
                        line.Visible = true
                        line.From = Vector2.new(CurrentCamera.ViewportSize.X / 2, CurrentCamera.ViewportSize.Y)
                        line.To = Vector2.new(pos.X, pos.Y)
                        line.Color = GetESPColor()
                    else
                        line.Visible = false
                    end
                else
                    line.Visible = false
                end
            end
        end
    end))
end

ProtectionTab:AddSection({ "Anti Void" })
local AntiVoidEnabled = false
local AntiVoidConn = nil
local AntiVoidCharConn = nil
local AntiVoidPositions = {}
local AntiVoidTask = nil
local function AntiVoidApply(character)
    local root = character:FindFirstChild("HumanoidRootPart")
    if not root then return end
    AntiVoidPositions = {}
    AntiVoidTask = task.spawn(function()
        while AntiVoidEnabled do
            if root and root.Parent then
                table.insert(AntiVoidPositions, { pos = root.Position, time = tick() })
                for i = #AntiVoidPositions, 1, -1 do
                    if tick() - AntiVoidPositions[i].time > 5 then
                        table.remove(AntiVoidPositions, i)
                    end
                end
                local y = root.Position.Y
                if y < -300 then
                    for i = #AntiVoidPositions, 1, -1 do
                        if tick() - AntiVoidPositions[i].time >= 3 then
                            root.CFrame = CFrame.new(AntiVoidPositions[i].pos)
                        end
                    end
                end
                if y < -460 then
                    root.CFrame = CFrame.new(root.Position.X, 100, root.Position.Z)
                end
                task.wait(0.1)
            end
        end
    end)
end
ProtectionTab:AddToggle({
    Name = "Anti Void",
    Default = false,
    Callback = function(value)
        AntiVoidEnabled = value
        if value then
            if LocalPlayer.Character then
                AntiVoidApply(LocalPlayer.Character)
            end
            AntiVoidCharConn = LocalPlayer.CharacterAdded:Connect(function(char)
                task.wait(0.2)
                if AntiVoidEnabled then
                    AntiVoidApply(char)
                end
            end)
        else
            AntiVoidEnabled = false
            if AntiVoidCharConn then
                AntiVoidCharConn:Disconnect()
                AntiVoidCharConn = nil
            end
            if AntiVoidTask then
                task.cancel(AntiVoidTask)
            end
        end
    end
})

ProtectionTab:AddSection({ "Anti Sit" })
local AntiSitGlobalEnabled = false
local AntiSitGlobalConn = nil
local AntiSitGlobalCharConn = nil
local function AntiSitGlobalApply(character)
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    if not humanoid then return end
    humanoid.Sit = false
    humanoid:SetStateEnabled(Enum.HumanoidStateType.Seated, false)
    if AntiSitGlobalConn then
        AntiSitGlobalConn:Disconnect()
    end
    AntiSitGlobalConn = humanoid:GetPropertyChangedSignal("Sit"):Connect(function()
        if AntiSitGlobalEnabled and humanoid.Sit then
            humanoid.Sit = false
            humanoid:ChangeState(Enum.HumanoidStateType.GettingUp)
        end
    end)
    task.spawn(function()
        while AntiSitGlobalEnabled and humanoid and humanoid.Parent do
            if humanoid.Sit then
                humanoid.Sit = false
                humanoid:ChangeState(Enum.HumanoidStateType.Running)
            end
            humanoid:SetStateEnabled(Enum.HumanoidStateType.Seated, false)
            task.wait()
        end
    end)
end

ProtectionTab:AddSection({ "Anti Skybox/Nuke" })
local AntiSkyboxEnabled = false
local AntiSkyboxPlayers = {}
local BodyIds = {
    ["96655874457685,123402086843885,78300682916056,86276701020724,78409653958165,120668655481073"] = "Body 1",
    ["100839513065432,123402086843885,78300682916056,86276701020724,78409653958165,120668655481073"] = "Body 2",
    ["123402086843885,100839513065432,78300682916056,86276701020724,78409653958165,15093053680"] = "Body 3",
    ["100839513065432,376531012,376530220,376531703,376531300,10638267973"] = "Body 4"
}
local function GetBodyKey(desc)
    return table.concat({ tostring(desc.Torso), tostring(desc.RightArm), tostring(desc.LeftArm), tostring(desc.RightLeg), tostring(desc.LeftLeg), tostring(desc.Head) }, ",")
end
local function MakeInvisible(character, visible)
    for _, part in pairs(character:GetDescendants()) do
        if part:IsA("BasePart") then
            part.LocalTransparencyModifier = visible and 0 or 1
        elseif part:IsA("Decal") then
            part.Transparency = visible and 0 or 1
        end
    end
end
local function CheckSkybox(player)
    if player == LocalPlayer then return end
    local char = player.Character
    if not char then return end
    local humanoid = char:FindFirstChildOfClass("Humanoid")
    if not humanoid then return end
    local success, desc = pcall(function() return humanoid:GetAppliedDescription() end)
    if not success or not desc then return end
    local key = GetBodyKey(desc)
    local isNuke = AntiSkyboxEnabled and BodyIds[key]
    if isNuke then
        MakeInvisible(char, false)
        if not AntiSkyboxPlayers[player] then
            AntiSkyboxPlayers[player] = true
            StarterGui:SetCore("SendNotification", {
                Title = "Nuke | Skybox",
                Text = player.Name .. " usando skybox | nuke " .. BodyIds[key],
                Duration = 5,
                Icon = "https://www.roblox.com/headshot-thumbnail/image?userId=" .. player.UserId .. "&width=420&height=420&format=png"
            })
        end
    else
        MakeInvisible(char, true)
        AntiSkyboxPlayers[player] = nil
    end
end
ProtectionTab:AddToggle({
    Name = "Anti Skybox | Nuke",
    Default = false,
    Callback = function(value)
        AntiSkyboxEnabled = value
        if not value then
            for _, player in pairs(Players:GetPlayers()) do
                if player.Character then
                    MakeInvisible(player.Character, true)
                end
            end
            AntiSkyboxPlayers = {}
        end
    end
})
RunService.RenderStepped:Connect(function()
    if AntiSkyboxEnabled then
        for _, player in pairs(Players:GetPlayers()) do
            CheckSkybox(player)
        end
    end
end)
Players.PlayerAdded:Connect(function(player)
    player.CharacterAdded:Connect(function()
        task.wait(2)
        CheckSkybox(player)
    end)
end)
